/**
 * Takes a weather code as input and returns a corresponding weather icon class.
 *
 * @param {number} weatherCode - A code representing the weather condition.
 * @returns {string} A string representing the weather icon class.
 */
function getWeatherIconClass(weatherCode) {
   switch (weatherCode) {
     case 0: // clear
       return "wi-day-sunny";
     case 1: // partly cloudy
     case 2: // cloudy
       return "wi-day-cloudy";
     case 3: // overcast
       return "wi-cloudy";
     case 45: // foggy
     case 48: // freezing fog
       return "wi-fog";
     case 51: // drizzle
     case 53: // light rain showers
     case 55: // moderate rain showers
     case 56: // light freezing drizzle
     case 57: // freezing drizzle
     case 58: // heavy freezing drizzle
     case 61: // light rain
     case 63: // moderate rain
     case 65: // heavy rain
     case 66: // light freezing rain
     case 67: // freezing rain
     case 68: // heavy freezing rain
       return "wi-rain";
     case 71: // light snow
     case 73: // moderate snow
     case 75: // heavy snow
     case 77: // snow grains
     case 80: // rain showers
     case 81: // heavy rain showers
     case 82: // ragged rain showers
       return "wi-showers";
     case 85: // light snow showers
     case 86: // heavy snow showers
       return "wi-snow";
     case 95: // thunderstorm
     case 96: // heavy thunderstorm
     case 99: // thunderstorm with hail
       return "wi-thunderstorm";
     default:
       return "wi-day-sunny"; // default to clear icon
   }
 }
 
/**
 * Updates the weather information on the page based on the current weather conditions.
 *
 * @param {boolean} farenheit - A boolean value indicating whether the temperature should be displayed in Farenheit or Celsius.
 */
function updateWeather(farenheit){
   fetch("https://api.open-meteo.com/v1/forecast?latitude=30.612214183103088&longitude=-96.34101092562297&current_weather=true&weathercode=true")
      .then((response) => response.json())
      .then((json) =>
         {
            document.getElementById("currTempF").innerHTML = (json.current_weather.temperature * 1.8 + 32).toFixed(1) + "F";
            document.getElementById("currTempC").innerHTML = json.current_weather.temperature.toFixed(1) + "C";
            switch(json.current_weather.weathercode){
               case 0:
               case 1:
                  document.getElementById("currWeather").innerHTML = "Clear";
                  break;
               case 2:
                  document.getElementById("currWeather").innerHTML = "Partly Cloudy";
                  break;
               case 3:
                  document.getElementById("currWeather").innerHTML = "Overcast";
                  break;
               case 45:
               case 48:
                  document.getElementById("currWeather").innerHTML = "Foggy";
                  break;
               case 51:
               case 53:
               case 55:
                  document.getElementById("currWeather").innerHTML = "Drizzle";
                  break;
               case 56:
               case 57:
                  document.getElementById("currWeather").innerHTML = "Freezing Drizzle";
                  break;
               case 61:
               case 63:
               case 65:
                  document.getElementById("currWeather").innerHTML = "Raining";
                  break;
               case 66:
               case 67:
                  document.getElementById("currWeather").innerHTML = "Freezing Rain";
                  break;
               case 71:
               case 73:
               case 75:
                  document.getElementById("currWeather").innerHTML = "Snowing";
                  break;
               case 77:
                  document.getElementById("currWeather").innerHTML = "Snow Grains";
                  break;
               case 80:
               case 81: 
               case 82:
                  document.getElementById("currWeather").innerHTML = "Rain Showers";
                  break;
               case 85:
               case 86:
                  document.getElementById("currWeather").innerHTML = "Snow Showers";
                  break;
               case 95:
               case 96:
               case 99:
                  document.getElementById("currWeather").innerHTML = "Thunderstorm";
                  break;
               default:
                  document.getElementById("currWeather").innerHTML = "Error";
                  break;
            }
            addIcon(getWeatherIconClass(json.current_weather.weathercode));
         }
      );
}
/**
 * Adds a weather icon to the HTML document.
 *
 * @param {string} url - The URL of the weather icon to add.
 * @returns {void}
 */
function addIcon(url) {
   const container = document.getElementById("iconContainer");
   if (container.firstChild) {
      container.removeChild(container.firstChild);
   }
   const img = document.createElement("i");
   img.classList.add('wi', url); // update to classList.add
   img.alt = "Weather icon";
   container.appendChild(img);
}
/**
 * Function to be executed when the window has finished loading.
 * It calls the `updateWeather` function to update the weather information.
 *
 * @function
 * @global
 */
window.onload = function(){
   updateWeather()
}
